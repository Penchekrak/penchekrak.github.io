Consider an alphabet of two letters, $\mathfrak{a}$ and $\mathfrak{b}$. Those will also be the two tokens in our language, so i will use tokens, letters and symbols interchangeably. Also there is $\emptyset$ which is the empty symbol and serve as a beginning-of-sequence as well.

$$ \mathcal{V} = \{\mathfrak{a}, \mathfrak{b}\}, c \in \mathcal{V}, \forall k \ s_k \in \mathcal{V} $$

Consider also a simple decoder-only causal language model with one parameter $\alpha$ that is defined as following Markov process:

$$ p(s_n = c \, \vert \, s_{n-1}, \dots, s_1) = \frac{1}{n}\left(\sum_{k = 1}^{n - 1}[s_k = c] + \alpha * [s_{n} = \mathfrak{a}] + (1- \alpha) * [s_{n} = \mathfrak{b}]\right) $$

at each step of the proper generation we will sample the next token from the distribution, or take the maximal probability token according to our strategy.

This LLM can also be written in vector form for the easy of use and implementation. $$ \forall n \ \mathbf{p}(s_{n-1}, \dots, s_1) \in I_{[0, 1]}{}^{2},\ \text{s.t.} \ \mathbf{p}_1 + \mathbf{p}_2 = 1 \\ \Rightarrow \mathbf{p}_1 = p(s_n = \mathfrak{a} \, \vert \, \dots), \ \mathbf{p}_2 = p(s_n = \mathfrak{b} \, \vert \, \dots) \\ {} \\ \text{OHE}(s_k) = \begin{pmatrix}[s_{k} = \mathfrak{a}] \\ [s_{k} = \mathfrak{b}]\end{pmatrix} $$

$$ \mathbf{p}(s_{n-1}, \dots, s_1) = \frac{1}{n}\left(\sum_{k = 1}^{n - 1}\text{OHE}(s_k) + \begin{pmatrix}\alpha \\ 1 - \alpha\end{pmatrix}\right) $$

Comparing the above formulation with our understanding of LM pipeline, we see that $\text{OHE}(s_k)$ is the embedding of the $k$-th token, and $\mathbf{p}(s_{n-1}, \dots, s_1)$ is the probability distribution of the next token. Only the bias term is tunable, and the main layer is a pooling of all the previous tokens embeddings (plus bias).

Behaviour of the model is pretty straightforward, the more particular symbol is frequent in the sequence, the more it is likely to be sampled. This is especially evident in the case of greedy sampling strategy.

$$ \text{Hardmax}(\mathbf{p}) = \begin{pmatrix}[p_{1} = \text{argmax}_k \mathbf{p}_k] \\ [p_{2} = \text{argmax}_k \mathbf{p}_k]\end{pmatrix} \\ {} \\ \text{OHE}(s_n) = \text{Hardmax}(\mathbf{p}(s_{n-1}, \dots, s_1)) = \text{Hardmax}\left(\frac{1}{n}\left(\sum_{k = 1}^{n - 1}\text{OHE}(s_k) + \begin{pmatrix}\alpha \\ 1 - \alpha\end{pmatrix}\right)\right) $$

import torch
torch.random.manual_seed(0xCAFFE)


class Hardmax(torch.nn.Module):
    def __init__(self, dim=-1):
        super().__init__()
        self.dim = dim

    def forward(self, input: torch.Tensor) -> torch.Tensor:
        return torch.nn.functional.one_hot(
            torch.argmax(input, dim=self.dim, keepdim=True),
            num_classes=input.shape[self.dim],
        )


class OneParameterLLM(torch.nn.Module):
    def __init__(self, logit_alpha: torch.Tensor, activation: torch.nn.Module):
        super().__init__()
        self.logit_alpha = torch.nn.Parameter(
            logit_alpha
        )  # we parametrize alpha in logits so optimization is unconstrained
        self.activation = activation

    def forward(self, input_embeddings: torch.Tensor) -> torch.Tensor:
        alpha = torch.sigmoid(
            self.logit_alpha
        )  # we parametrize alpha in logits so optimization is unconstrained
        return self.activation(
            (
                torch.sum(input_embeddings, dim=0)
                + torch.stack([alpha, 1 - alpha], dim=0)
            )
            / (input_embeddings.shape[0] + 1)
        )


class Tokenizer:
    def __init__(self):
        self.token_to_index = {"a": 0, "b": 1}
        self.index_to_token = {0: "a", 1: "b"}

    def to_embeddings(self, string: str) -> int:
        if string == "":
            return torch.empty(0, 2)
        return torch.stack(
            [
                torch.nn.functional.one_hot(
                    torch.tensor(self.token_to_index[token]), num_classes=2
                )
                for token in string
            ],
            dim=0,
        )

    def from_hard_sampler_output(self, index: torch.Tensor) -> str:
        return self.index_to_token[index.item()]


class GreedySampler:
    def __init__(self) -> None:
        pass

    def sample(self, probs: torch.Tensor) -> torch.Tensor:
        return torch.argmax(probs, dim=-1)


class WeightedSampler:
    def __init__(self) -> None:
        pass

    def sample(self, probs: torch.Tensor) -> torch.Tensor:
        return torch.multinomial(probs, num_samples=1)

Now let's formulate toy adversarial loss. Suppose our segmentation breaks when input contains a lot of $\mathfrak{b}$ symbols. Starting string is good tho and contains a lot of $\mathfrak{a}$'s.

$$ \mathcal{L} = - \frac{1}{n} \sum_{k = 1}^{n} [s_k = \mathfrak{b}] = - \frac{1}{n} \sum_{k = 1}^{n} \text{OHE}(s_k)_2 $$

First of all such loss is non-differentiable, like in our case where loss is dependent on particular tokens in the adversarial prompt. But we can make approximation of it by using probabilities instead of hard assignments.

$$ \hat{\mathcal{L}} = - \frac{1}{n} \sum_{k = 1}^{n} p(s_k = \mathfrak{b} \, \vert \, s_{k-1}, \dots, s_1) $$

Other gradient killing component is our sampling strategy. In either Greedy or Weighted sampling we are using non-smooth operations (namely argmax and multinomial sampling). We can use smooth approximations of those operations, like softmax and Gumbel-Softmax and in this manner pass weighted mixture of embeddings into next step of sequence generation instead of materialized token. Notice that in our case, since embeddings are just one-hot-encoded, such mixture is probability vector as is.

def softloss(probs: list[torch.Tensor]) -> torch.Tensor:
    return -sum(prob[1] for prob in probs) / len(probs)


class SoftmaxSampler:
    def __init__(self) -> None:
        pass

    def sample(self, probs: torch.Tensor) -> torch.Tensor:
        return probs  # just return probs as is. Real lm outputs not probs but logits, soi called this softmax.

    # This is a smooth approximation of argmax and in turn an approximation of greedy sampling


class GumbelSoftmaxSampler:
    def __init__(self) -> None:
        pass

    def sample(self, probs: torch.Tensor) -> torch.Tensor:
        return torch.nn.functional.gumbel_softmax(
            torch.log(probs), dim=-1
        )  # this only works for lm with identity activation as we need original probs to be passed

from matplotlib import pyplot as plt
from IPython.display import clear_output


def generate(initial_string, lm, maxlen, sampler):
    string = initial_string
    prob_history = []
    embeddings = tokenizer.to_embeddings(string)
    for i in range(maxlen):
        next_token_probs = lm(embeddings)
        prob_history.append(next_token_probs.squeeze())
        sampled_probs = sampler.sample(next_token_probs)
        next_token_embedding = sampled_probs @ torch.eye(2)
        embeddings = torch.cat([embeddings, next_token_embedding.unsqueeze(0)], dim=0)
        next_token = tokenizer.from_hard_sampler_output(sampled_probs.argmax(dim=-1))
        string += next_token
    return string, prob_history


logit_alpha = torch.logit(torch.tensor(0.75))  # so alpha = 0.75

from collections import defaultdict


def denis_way(lm, sampler, initial_string, maxlen):
    string, _ = generate(initial_string, lm, maxlen, sampler)
    prob_history = []
    for i in range(maxlen):
        embeddings = tokenizer.to_embeddings(string[:i])
        current_prob = lm(embeddings)
        prob_history.append(current_prob.squeeze())
    loss = softloss(prob_history)
    return loss


sampler = SoftmaxSampler()

alphas = torch.linspace(0, 1, 100)
logit_alphas = torch.logit(alphas)
results = defaultdict(list)
for logit_alpha in logit_alphas:
    lm.logit_alpha.data = logit_alpha
    results[logit_alpha].append(denis_way(lm, sampler, initial_string, 10).item())

plt.plot(alphas, [torch.mean(torch.tensor(losses)) for losses in results.values()])
